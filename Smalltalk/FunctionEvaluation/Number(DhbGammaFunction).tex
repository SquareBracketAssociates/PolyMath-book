{\bf gamma}
\begin{verbatim}
    ^self > 1
        ifTrue: [ ^DhbLanczosFormula new gamma: self]
        ifFalse:[ self < 0
                        ifTrue: [ Float pi / ( ( Float pi * self) sin 
                                                 * ( 1 - self) gamma)]
                        ifFalse:[ ( DhbLanczosFormula new gamma: ( 
                                                    self + 1)) / self]
                    ]

\end{verbatim}
